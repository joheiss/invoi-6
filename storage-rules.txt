service firebase.storage {
  match /b/{bucket}/o {
  	match /docs/{allPaths=**} {
      allow read, write: if request.auth != null && hasRole('sales-user');
    }

    match /assets/{allPaths=**} {
      allow read: if request.auth != null;
      allow write: if request.auth != null && hasRole('sys-admin');
    }

    function getUserData() {
      return get(/databases/$(database)/documents/user-profiles/$(request.auth.uid)).data
    }

    function getUserRoles() {
      return getUserData().roles;
    }

    function hasRole(role) {
      return getUserRoles().hasAny([role]);
    }
	}
}

match /{allPaths=**} {
  		allow read, write: if request.auth != null;
 		}
