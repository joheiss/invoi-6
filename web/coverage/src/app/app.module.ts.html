export function clearState(reducer: ActionReducer<AppState>): ActionReducer<AppState> {
  return function(state: AppState, action: Action): AppState {
    if (action.type === '[Auth] clear state') {
        state = undefined;
    }
    return reducer(state, action);
  };
}
